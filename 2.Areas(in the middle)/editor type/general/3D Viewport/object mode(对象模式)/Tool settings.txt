左侧工具栏:
    select:
        无

    select box:
        new 
        add 
        subtract
        difference
        intersect

    select circle:
        radius


    select lasso:
        new 
        add 
        subtract
        difference
        intersect

    cursor:
        surface project
        
        orientation:
            none
            view
            transform
            geometry
            orientation
